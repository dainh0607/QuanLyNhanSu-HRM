using System;
using System.Linq;
using Microsoft.EntityFrameworkCore;
using Microsoft.Extensions.Configuration;
using Microsoft.Extensions.DependencyInjection;
using Microsoft.Extensions.Hosting;
using ERP.Entities;

    .AddJsonFile("appsettings.json")
    .Build();

var optionsBuilder = new DbContextOptionsBuilder<AppDbContext>();

using (var db = new AppDbContext(optionsBuilder.Options))
{
    var user = db.Users.FirstOrDefault(u => u.email == "kfrog1233@gmail.com");
    if (user != null)
    {
        user.role = "Admin";
        db.SaveChanges();
        Console.WriteLine("SUCCESS: Updated user kfrog1233@gmail.com to Admin.");
    }
    else
    {
        Console.WriteLine("ERROR: User kfrog1233@gmail.com not found in database.");
    }
